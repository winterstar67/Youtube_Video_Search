youtube_video_infos와 result 에 해당 함수를 적용하도록 수정해줘.

        99 +      """
       100 +      Concatenate previous data and current data based on file extension
       101 +  
       102 +      Args:
       103 +          prev_data: Previous data (DataFrame for csv, dict for json)
       104 +          current_data: Current data (DataFrame for csv, dict for json)
       105 +          file_extension: File extension ('csv' or 'json')
       106 +  
       107 +      Returns:
       108 +          Concatenated data in the same format as input
       109 +  
       110 +      Example:
       111 +          # For CSV
       112 +          >>> prev_df = pd.DataFrame({'id': [1], 'name': ['A']})
       113 +          >>> current_df = pd.DataFrame({'id': [2], 'name': ['B']})
       114 +          >>> result = concat_data(prev_df, current_df, 'csv')
       115 +          # result: DataFrame with rows [1, 'A'] and [2, 'B']
       116 +  
       117 +          # For JSON
       118 +          >>> prev_dict = {'key1': 'value1'}
       119 +          >>> current_dict = {'key2': 'value2'}
       120 +          >>> result = concat_data(prev_dict, current_dict, 'json')
       121 +          # result: {'key1': 'value1', 'key2': 'value2'}
       122 +      """
       123 +      if file_extension == 'csv':
       124 +          # CSV: concatenate along axis=0 (rows)
       125 +          return pd.concat([prev_data, current_data], axis=0, ignore_index=True)
       126 +      elif file_extension == 'json':
       127 +          # JSON: merge dictionaries (current_data overwrites prev_data for duplicate keys)
       128 +          merged = prev_data.copy()
       129 +          merged.update(current_data)
       130 +          return merged
       131 +      else:
       132 +          raise ValueError(f"Unsupported file extension: {file_extension}. Only 'csv' and 'json' are supported.")
